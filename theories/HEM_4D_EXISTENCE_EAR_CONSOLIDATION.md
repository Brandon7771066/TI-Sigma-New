def pd_scale(value: float, min_val: float, max_val: float) -> float:
    """Map to (-3, 2) PD scale."""
    normalized = (value - min_val) / (max_val - min_val)
    return (normalized * 5) - 3  # Maps [0,1] to [-3, 2]
