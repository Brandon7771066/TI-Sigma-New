## Part 4: Data Analysis Methods

### 4.1 Symmetry Scoring Algorithm

```python
import numpy as np
from PIL import Image

def calculate_8fold_symmetry(image_path):
    """
    Analyze Bio-Well image for 8-fold rotational symmetry
    Returns score from 0 (no symmetry) to 1 (perfect 8-fold)
    """
    # Load image
    img = Image.open(image_path).convert('L')
    img_array = np.array(img)
    
    # Find center (brightest point or image center)
    center_y, center_x = np.unravel_index(
        np.argmax(img_array), img_array.shape
    )
    
    # Extract 8 radial sectors
    sectors = []
    for i in range(8):
        angle_start = i * 45 - 22.5
        angle_end = i * 45 + 22.5
        sector = extract_sector(img_array, center_x, center_y, 
                               angle_start, angle_end)
        sectors.append(np.mean(sector))
    
    # Calculate symmetry score
    # Perfect 8-fold: all sectors equal
    # Score = 1 - normalized standard deviation
    sectors = np.array(sectors)
    if np.mean(sectors) > 0:
        symmetry_score = 1 - (np.std(sectors) / np.mean(sectors))
    else:
        symmetry_score = 0
    
    return max(0, min(1, symmetry_score))

def extract_sector(img, cx, cy, angle_start, angle_end, radius=100):
    """Extract pixels within angular sector"""
    y, x = np.ogrid[:img.shape[0], :img.shape[1]]
    angles = np.degrees(np.arctan2(y - cy, x - cx))
    distances = np.sqrt((x - cx)**2 + (y - cy)**2)
    
    # Handle angle wraparound
    mask = (angles >= angle_start) & (angles < angle_end)
    mask &= (distances < radius) & (distances > 10)  # Ring, not disk
    
    return img[mask]
```
